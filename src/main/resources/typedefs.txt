# Maps C/C++ types to Java/JNA

# Primitives
int8_t 					-> byte
uint8_t 				-> byte
int32_t 				-> int
uint32_t 				-> int
int64_t 				-> long
uint64_t 				-> long
size_t 					-> long

# Pointers
void* 					-> com.sun.jna.Pointer
char* 					-> String
char**					-> com.sun.jna.Pointer

# Arrays
char[]					-> byte

# Over-rides
VkBool32				-> org.sarge.jove.platform.vulkan.VulkanBoolean
